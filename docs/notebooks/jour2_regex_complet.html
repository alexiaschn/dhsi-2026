def trouver_racine_projet() -> Path:
    """Trouve la racine du projet depuis la racine ou depuis notebooks/."""
    cwd = Path.cwd().resolve()

    # Le notebook peut être lancé depuis plusieurs dossiers.
    # On remonte donc les dossiers parents jusqu'a trouver la structure du projet.
    for dossier in (cwd, *cwd.parents):
        if (dossier / "notebooks").exists() and (dossier / "corpus").exists():
            return dossier

    # Si la structure attendue n'est pas trouvée, on utilise le dossier courant.
    return cwd

expressions régulières de base

Cheat Sheet

. : n’importe quel caractère

 : précède un caractère que l’on souhaite “échapper”. Ex: re.search(“.”, texte) va chercher le premier point du texte tandis que re.search(“.”, texte) sélectionnera le premier caractère du texte.

Quelques caractères échappés

un chiffre : non chiffre : une lettre : non lettre

Ex : re.search(“d , texte) sélectionne la première instance du motif”d 1” ou “d 2” ou “d 3” etc.

^ : début d’une chaîne. frontière de mot : espace $ : fin de chaîne : saut de ligne tabulation

() : effectuer une capture ex: re.search(r”date: ()“,”date: 2026”) retourne “2026” | : équivalent de OU exemple re.search(r”(chien|chat)“,”j’aime les chiens”) retourne “chien”. [] : range, correspond toujours à un caractère ex: re.search(r”[A-Z]“,”mon nom est Alexia”) retourne “Alexia”. Exemple 2 : re.search(r”[abc].”, “magie”) retourne “ag” (a ou b ou c suivi de n’importe quel caractère).

Les quantifieurs

Suivent un caractère pour quantifier le nombre de fois où il apparait.

  • : 0 ou n fois
  • : 1 ou n fois ? : 0 ou 1 fois

Module re en Python

Documentation

re.match(pattern, string) -> match : sélectionne à partir du début de la chaîne re.search(pattern, string) -> match : sélectionne le motif n’importe où dans la chaîne (s’arrête à la première occurrence) re.findall(pattern, string) -> list : sélectionne toutes les instances du motif dans la chaîne re.split(pattern, string) -> list : divise la chaine selon le motif regex re.sub(pattern, replacement_pattern, string) -> string : remplace un motif par un autre

match.group(0) : retourne toute la séquence du motif identifiée dans le texte match.group(1) : retourne le premier groupe capture

\1 = match.group(1) \2 = match.group(2) etc.

Nettoyer un corpus

Nettoyer un corpus signifie en général : supprimer les espaces supplémentaires, uniformiser la syntaxes, renommer les documents.

Prenons en exemple un corpus non nettoyé

épigrammes :

# importer les épigrammes avec l'API de l'Anthologie Grecque
from urllib.error import HTTPError, URLError
from urllib.request import Request, urlopen

EPIGRAMMES = [
    "7.324",
    "7.447",
    "7.381",
]
API_TEMPLATE = "https://anthologiagraeca.org/api/passages/urn:cts:greekLit:tlg7000.tlg001.ag:{epigramme_id}/"


def charger_passage(epigramme_id: str) -> dict:
    """Appelle l'API pour une épigramme et renvoie sa reponse JSON."""
    url = API_TEMPLATE.format(epigramme_id=epigramme_id)

    # Les en-têtes indiquent que l'on souhaite recevoir du JSON.
    requete = Request(
        url,
        headers={
            "Accept": "application/json",
            "User-Agent": "dhsi-2026-corpus/1.0",
        },
    )
    try:
        with urlopen(requete, timeout=30) as reponse:
            return json.load(reponse)
    except (HTTPError, URLError) as erreur:
        raise RuntimeError(f"Erreur pendant le chargement de {epigramme_id}: {erreur}") from erreur


corpus = []
for epigramme_id in EPIGRAMMES:
    passage = charger_passage(epigramme_id)
    corpus.append(passage)

def nettoyer_texte(texte: str) -> str:
    """Nettoie le texte français récupéré depuis l'API."""
    # Recolle les mots coupés par un tiret suivi d'un espace ou d'un retour à la ligne.
    # Exemple : "sau- terelle" devient "sauterelle".
    texte = re.sub(r"(?<=\w)-\s+(?=\w)", "", texte)

    # Transforme ensuite les retours à la ligne et espaces multiples en espaces simples.
    return re.sub(r"\s+", " ", texte).strip()
